# WealthLog Quick Fix Commands
# Run these commands one by one in your terminal

# Navigate to shared package
cd C:\Users\User\Desktop\wealthlogs\wealthlogs-code\packages\shared

# Clean everything
rmdir /s /q node_modules
del /f package-lock.json
rmdir /s /q dist

# Install dependencies fresh
npm install

# Explicitly install @types/node
npm install --save-dev @types/node@20.0.0

# Try to build
npm run build

# If build fails, try direct compilation
npx tsc

# If still failing, try with no emit on error disabled
npx tsc --noEmitOnError false

# Alternative: Build with JavaScript output only (no type checking)
npx tsc --noEmit false --skipLibCheck true

# Once shared package builds, continue with other packages:

# Backend
cd ..\..\apps\backend
npm install --legacy-peer-deps
npx prisma generate

# Frontend
cd ..\web
npm install --legacy-peer-deps

# Root
cd ..\..
npm install --legacy-peer-deps
